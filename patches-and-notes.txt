Ensure:

sudo virsh net-autostart default

Plocate:

sudo mkdir -p /var/lib/plocate
sudo chmod 755 /var/lib/plocate

For iGPU passthrough on main desktop only:

sudo rpm-ostree kargs --append="amd_iommu=on iommu=pt"
sudo rpm-ostree kargs --append="rd.driver.pre=vfio-pci"
systemctl reboot
sudo rpm-ostree kargs --append="vfio-pci.ids=1002:164e,1002:1640"
systemctl reboot
